Javascript 倒计时

2022-07-11 Javascript 
<script>

function Timing(idname, time, obj, msg){
    var t=eval(parseInt(time)+1);//+1,初始页面显示time，而非(time-1);
    document.getElementById(idname).innerHTML=t;
    function Countdown(){
        if(t>0){
        t--;
        document.getElementById(idname).innerHTML=t
        var action=setTimeout(Countdown,1000);
        }else{
            clearTimeout(action);
            document.getElementById(idname).parentNode.innerHTML=msg;
            }
    };
    Countdown();
};

Timing('idname',30,"Countdown","号码已回收！");
